<objective>
Implement the `VisibilityProjector` (D-14) — the pure-function projection of graph state for an actor that Phase 5's observer synthesizes from. This closes **GAP-CROSS01** ("highest-leverage cross-cutting gap" per GAP-HANDOFF.md) and implements the minimum belief representation (GAP-GRAPH04) needed for MECH10 `tell` and MECH25 `partial_knowledge_update` to behave correctly.

The projector composes four operations in order:
1. **Containment walk** — start at actor; add actor.location (room); add room's contained nodes (via `inside` / `on` / `contains` edges); add actor's held items (`holds` edge).
2. **Illumination filter** — if a room's `illumination` < threshold AND actor doesn't hold a `light_source`-tagged entity, dim the room (remove contained-node detailed properties, show only a darkness marker).
3. **Property visibility classes** — for each projected node, filter out properties listed in that node's `hidden_properties` list (if present).
4. **Belief overlay** — if `actor.beliefs[node_id]` exists for a node already in projection, overlay those key/value pairs on top of ground truth. Beliefs for nodes NOT in the projection are silently ignored (no phantom entities).

The output: a JSON-serializable `dict[str, dict]` keyed by node id, each value = `{"type": ..., "properties": {...}, "edges": [...]}`. This dict is what the Sonnet observer (Plan 05-05) consumes with a hard grounding constraint — the observer can only mention what's in this dict.
</objective>

"""VisibilityProjector (D-14) — pure-function projection of graph state for an actor.

Closes GAP-CROSS01 (observation projection) and GAP-GRAPH04 (belief vs ground
truth). Output is a JSON-serializable dict[node_id, entry] consumed by the
Sonnet observer under a hard grounding constraint.

Composition (in order):
    1. Containment walk  — actor, location, contained neighbours, held items
    2. Illumination filter — dim rooms unless actor holds a light_source
    3. Property visibility — honour per-node `hidden_properties`
    4. Belief overlay     — actor.beliefs overrides ground truth for already-projected nodes
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from token_world.graph import KnowledgeGraph

ILLUMINATION_THRESHOLD = 0.2  # nodes in rooms below this need light source to see details


@dataclass(slots=True)
class VisibilityProjector:
    """Project graph state for one actor's observation (D-14)."""
    graph: KnowledgeGraph

    def project_for(self, actor_id: str) -> dict[str, dict[str, Any]]:
        if not self.graph.has_node(actor_id):
            return {}
        projection: dict[str, dict[str, Any]] = {}

        # Stage 1: containment walk
        self._add_node(projection, actor_id)
        location_id = self._get_actor_location(actor_id)
        if location_id is not None and self.graph.has_node(location_id):
            self._add_node(projection, location_id)
            # Add nodes contained inside the location
            for contained in self._neighbors_by_edge_types(location_id, ["contains", "inside", "on"]):
                self._add_node(projection, contained)
        # Held items
        for held in self._neighbors_by_edge_types(actor_id, ["holds"]):
            self._add_node(projection, held)

        # Stage 2: illumination filter
        projection = self._apply_illumination_filter(projection, actor_id, location_id)

        # Stage 3: property visibility classes
        projection = self._apply_hidden_properties(projection)

        # Stage 4: belief overlay
        projection = self._apply_belief_overlay(projection, actor_id)

        return projection

    def _add_node(self, projection: dict, node_id: str) -> None:
        if node_id in projection:
            return
        props = self.graph.query(node_id) or {}
        node_type = props.get("type", "entity")
        edges = [
            {"type": e_type, "target": target}
            for (e_type, target) in self._outgoing_edges(node_id)
        ]
        projection[node_id] = {
            "type": node_type,
            "properties": dict(props),  # defensive copy
            "edges": edges,
        }

    def _get_actor_location(self, actor_id: str) -> str | None:
        for e_type, target in self._outgoing_edges(actor_id):
            if e_type == "location":
                return target
        return None

    def _neighbors_by_edge_types(self, node_id: str, edge_types: list[str]) -> list[str]:
        """Return target node IDs connected to node_id via any of edge_types."""
        return [target for e_type, target in self._outgoing_edges(node_id) if e_type in edge_types]

    def _outgoing_edges(self, node_id: str) -> list[tuple[str, str]]:
        """Return list of (edge_type, target_node_id) for outgoing edges.

        Inspect KnowledgeGraph API; if a direct helper doesn't exist, fall back
        to iterating all edges and filtering.
        """
        # Implementation depends on actual KG API. Pseudo:
        edges = []
        for target in self.graph.neighbors(node_id):
            # Get edge type; KG stores edges with properties, one of which is 'type'
            edge_props = self.graph.query_edge(node_id, target) if hasattr(self.graph, "query_edge") else {}
            e_type = edge_props.get("type", "related")
            edges.append((e_type, target))
        return edges

    # Stubs — filled in Tasks 2/3/4
    def _apply_illumination_filter(self, projection, actor_id, location_id): return projection
    def _apply_hidden_properties(self, projection): return projection
    def _apply_belief_overlay(self, projection, actor_id): return projection
```

def _apply_illumination_filter(self, projection, actor_id, location_id) -> dict:
    if location_id is None or location_id not in projection:
        return projection
    room_entry = projection[location_id]
    illumination = room_entry["properties"].get("illumination", 1.0)
    if illumination >= ILLUMINATION_THRESHOLD:
        return projection

    # Check actor's held items for a light source
    actor_entry = projection.get(actor_id, {})
    actor_edges = actor_entry.get("edges", [])
    held_ids = [e["target"] for e in actor_edges if e["type"] == "holds"]
    for held in held_ids:
        held_entry = projection.get(held, {})
        held_props = held_entry.get("properties", {})
        held_tags = held_props.get("tags", []) or []
        if "light_source" in held_tags or held_props.get("light_source") is True:
            # Actor has light — no dimming.
            return projection

    # Dim: strip contained-node properties, mark room as dimmed.
    # Contained nodes are those connected to location via contains/inside/on edges.
    contained_ids = {
        e["target"] for e in room_entry.get("edges", [])
        if e["type"] in ("contains", "inside", "on")
    }
    new_projection = {}
    for node_id, entry in projection.items():
        if node_id == location_id:
            # Mark room dimmed
            dimmed_entry = dict(entry)
            dimmed_entry["dimmed"] = True
            new_projection[node_id] = dimmed_entry
        elif node_id in contained_ids:
            # Strip detailed properties
            new_projection[node_id] = {
                "type": entry["type"],
                "properties": {},  # details invisible in dark
                "edges": [],
                "dimmed": True,
            }
        else:
            # Actor + held items unchanged
            new_projection[node_id] = entry
    return new_projection
```

def _apply_hidden_properties(self, projection) -> dict:
    new_projection = {}
    for node_id, entry in projection.items():
        props = entry.get("properties", {})
        hidden_keys = props.get("hidden_properties", []) or []
        if hidden_keys:
            filtered_props = {k: v for k, v in props.items() if k not in hidden_keys}
            filtered_props.pop("hidden_properties", None)
            new_entry = dict(entry)
            new_entry["properties"] = filtered_props
            new_projection[node_id] = new_entry
        else:
            new_projection[node_id] = entry
    return new_projection
```

def _apply_belief_overlay(self, projection, actor_id) -> dict:
    actor_entry = projection.get(actor_id)
    if actor_entry is None:
        return projection
    beliefs = actor_entry.get("properties", {}).get("beliefs", {}) or {}
    if not isinstance(beliefs, dict) or not beliefs:
        return projection

    new_projection = dict(projection)
    for node_id, believed_props in beliefs.items():
        if node_id not in new_projection:
            continue  # Pitfall 15: no phantoms
        if not isinstance(believed_props, dict):
            continue
        merged_entry = dict(new_projection[node_id])
        merged_props = dict(merged_entry.get("properties", {}))
        merged_props.update(believed_props)  # beliefs override
        merged_entry["properties"] = merged_props
        new_projection[node_id] = merged_entry
    return new_projection
```
