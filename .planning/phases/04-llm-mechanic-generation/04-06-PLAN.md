Step A — Add helper to `src/token_world/mechanic/seeds/_helpers.py`:
```python
"""Shared helpers for seed mechanics. Registry skips _*.py files (D-05)."""
from __future__ import annotations

from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from token_world.mechanic.context import MechanicContext


def _find_open_passage(ctx: "MechanicContext", src: str, dst: str) -> str | None:
    """Return the id of an open passage entity connecting src to dst, or None.

    A passage is an entity with subtype='doorway' (or 'passage') that sits
    between src and dst via 'connects' edges and has `open=True`.
    Shared with `try_door` (plan 04-07 MECH27).
    """
    for neighbor in ctx.neighbors(src, relation="connects"):
        props = ctx.query_node(neighbor)
        subtype = props.get("subtype")
        if subtype not in {"doorway", "passage"}:
            continue
        if not props.get("open", False):
            continue
        # passage → dst via connects?
        if dst in set(ctx.neighbors(neighbor, relation="connects")):
            return neighbor
    return None
```
(If `ctx.neighbors` signature takes `relation=` differently — check `MechanicContext` — adapt accordingly. If no filter supported, use `query_neighbors` + filter by props.)

```python
"""MECH01 passage_move seed: agent moves through a passage entity."""
from __future__ import annotations

from typing import TYPE_CHECKING

from token_world.graph import Mutation
from token_world.mechanic.protocol import CheckResult, Mechanic
from token_world.mechanic.seeds._helpers import _find_open_passage

if TYPE_CHECKING:
    from token_world.mechanic.context import MechanicContext


class PassageMoveMechanic(Mechanic):
    """Agent moves from current location to a connected location through a doorway/passage.

    Preconditions:
        - Actor exists and has a located_in edge to some src
        - Target location exists
        - An entity P with subtype in {doorway, passage} sits between src and target with P.open=True

    Side effects:
        - Remove actor's located_in edge to src
        - Add located_in edge to target
    """

    id = "passage_move"
    description = "Agent moves through an open passage entity"
    voluntary = True
    tags: list[str] = ["spatial", "passage"]

    def check(self, ctx: "MechanicContext") -> CheckResult:
        if not ctx.has_node(ctx.actor):
            return CheckResult(passed=False, reasons=["actor does not exist"])
        if not ctx.has_node(ctx.target):
            return CheckResult(passed=False, reasons=["target does not exist"])
        # Find actor's current location
        src = _current_location(ctx, ctx.actor)
        if src is None:
            return CheckResult(passed=False, reasons=["actor has no located_in location"])
        if src == ctx.target:
            return CheckResult(passed=False, reasons=["already at target"])
        passage = _find_open_passage(ctx, src, ctx.target)
        if passage is None:
            return CheckResult(passed=False, reasons=["no open passage to target"])
        return CheckResult(passed=True)

    def apply(self, ctx: "MechanicContext") -> list[Mutation]:
        src = _current_location(ctx, ctx.actor)
        muts: list[Mutation] = []
        if src is not None:
            muts.append(ctx.remove_edge(ctx.actor, src, relation="located_in"))
        muts.append(ctx.add_edge(ctx.actor, ctx.target, relation="located_in"))
        return muts


def _current_location(ctx: "MechanicContext", actor: str) -> str | None:
    """Return the actor's current located_in target, or None."""
    for n in ctx.neighbors(actor, relation="located_in"):
        return n
    return None
```
