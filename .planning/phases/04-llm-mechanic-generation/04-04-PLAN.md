Step C — Create `tests/test_integration/test_use_cases.py`:

```python
"""Parametrized integration tests driven by Phase 3 use-case manifests (D-26..D-29b).

Outcome branching (W5 — explicit):
- 'blocked' → pytest.skip(reason=gap_id)
- 'yield'   → pytest.xfail(reason=...) when no mechanic fires; run assertions when one does
- 'pass'    → pytest.fail(...) when no mechanic fires; run full assertions otherwise

Import safety (W9): _discover_cases() is wrapped; any exception at module-import time
collapses to a single skipped test case rather than poisoning pytest collection.
"""
from __future__ import annotations

import re
from pathlib import Path
from typing import Any

import pytest

from token_world.mechanic import ChainExecutionEngine, MechanicRegistry
from token_world.use_cases.loader import load_use_case, validate_frontmatter

from .conftest import CATEGORIES, SEEDS_DIR, USE_CASES_DIR


def _classify_outcome(fm: dict) -> tuple[str, str | None]:
    """Return (outcome, reason) where outcome in {'pass','yield','blocked'}.

    Default: 'pass' if field absent.
    For 'blocked': reason = first engine-layer address-now gap id found, or generic.
    """
    outcome = fm.get("expected_outcome") or "pass"
    if outcome == "blocked":
        # Try to derive a concrete gap ID from gaps list
        for gap in fm.get("gaps", []) or []:
            if gap.get("layer") == "engine" and gap.get("severity") == "address-now":
                # Gap id may not be present in the manifest's inline gaps (they're free-form).
                # Use summary as the reason string.
                return outcome, gap.get("summary", "engine-layer framework gap")
        return outcome, "framework gap"
    return outcome, None


def _discover_cases() -> list:
    """Collect pytest.param entries for every use-case manifest.

    Per W9: this function is called ONCE at module import time. Any exception
    here would poison test collection for the entire file. The caller
    (_PARAMS assignment below) wraps this call in try/except so that a
    catastrophic failure degrades to a single skipped test rather than an
    ImportError.
    """
    params: list[Any] = []
    for category in CATEGORIES:
        cat_dir = USE_CASES_DIR / category
        if not cat_dir.is_dir():
            continue
        for path in sorted(cat_dir.glob("UC-*.md")):
            try:
                fm, _body = load_use_case(path)
            except Exception as e:
                # Broken manifest — surface as test collection SKIP, do NOT raise
                params.append(pytest.param(
                    path,
                    id=f"{path.stem}-LOAD-ERROR",
                    marks=pytest.mark.skip(reason=f"manifest load failed: {e}"),
                ))
                continue
            errs = validate_frontmatter(fm, source=str(path))
            if errs:
                params.append(pytest.param(
                    path,
                    id=f"{path.stem}-SCHEMA-ERROR",
                    marks=pytest.mark.skip(reason="; ".join(errs)),
                ))
                continue
            uc_id = fm["id"]
            # Collection-time markers are a soft hint; the test body does
            # the authoritative branching (W5).
            params.append(pytest.param(path, id=uc_id))
    return params


# W9: import-safe discovery. If _discover_cases itself raises (e.g.,
# USE_CASES_DIR is missing in a weird CI environment), fall back to a single
# skipped param so pytest collection doesn't fail with ImportError.
try:
    _PARAMS = _discover_cases()
    if not _PARAMS:
        _PARAMS = [pytest.param(
            Path("(no-manifests-found)"),
            id="no-manifests-found",
            marks=pytest.mark.skip(reason="no use-case manifests discovered"),
        )]
except Exception as _disc_err:  # pragma: no cover — defensive
    _PARAMS = [pytest.param(
        Path("(discovery-failed)"),
        id="discovery-failed",
        marks=pytest.mark.skip(reason=f"_discover_cases raised: {_disc_err}"),
    )]


@pytest.mark.parametrize("use_case_path", _PARAMS)
def test_use_case(
    use_case_path: Path,
    harness_kg,
    diagnostics_sink,  # W2: exercise D-23 sink on every test
) -> None:
    """Run a use case: build graph, invoke mechanic chain, assert expected state.

    Outcome branching is explicit (W5) — the test body decides xfail/fail/skip
    per resolved outcome, rather than relying on collection-time markers alone.
    """
    fm, _body = load_use_case(use_case_path)
    uc_id = fm["id"]
    outcome, blocked_reason = _classify_outcome(fm)

    # Start a diagnostics tick BEFORE any work — W2: every run writes a summary.
    # Use hash of uc_id modulo a reasonable range as a pseudo tick-id; this is
    # per-test-isolated (tmp_path-rooted sink), so collisions are not a concern.
    pseudo_tick = abs(hash(uc_id)) % 100000
    with diagnostics_sink.open_tick(pseudo_tick) as tick_ctx:
        tick_ctx.write_action(f"UC harness: {uc_id}")
        mechanics_fired: list[str] = []

        # Early branch: blocked use cases skip BEFORE running the engine.
        if outcome == "blocked":
            tick_ctx.set_summary(
                uc_id=uc_id,
                outcome="blocked",
                mechanics_fired=[],
                reason=blocked_reason or "framework gap",
            )
            pytest.skip(f"{uc_id}: blocked by {blocked_reason or 'framework gap'}")

        # 1. Build graph from setup.graph_builder
        builder_src = fm["setup"]["graph_builder"]
        try:
            exec(
                compile(builder_src, str(use_case_path), "exec"),
                {"kg": harness_kg},
            )
        except Exception as e:  # Pitfall 3: don't muddle setup errors with test failures
            tick_ctx.set_summary(uc_id=uc_id, outcome="error", mechanics_fired=[])
            pytest.fail(
                f"{uc_id}: setup.graph_builder is invalid: {type(e).__name__}: {e}"
            )

        # 2. Register stock seed mechanics
        registry = MechanicRegistry(SEEDS_DIR)
        engine = ChainExecutionEngine(harness_kg, registry)

        # 3. For each action, try to match + execute a voluntary mechanic
        actions = fm.get("actions") or []
        any_mechanic_fired = False
        for action in actions:
            actor = action["actor"]
            classified = action.get("classified") or {}
            verb = classified.get("verb")
            target = classified.get("target") or classified.get("indirect_object")
            if not verb or not target:
                continue
            # Naive matching for v1: find a voluntary mechanic whose id == verb.
            # Phase 5 replaces this with proper classifier→mechanic routing.
            matched_id = None
            for info in registry.list_mechanics():
                if info.voluntary and info.id == verb:
                    matched_id = info.id
                    break
            if matched_id is None:
                continue
            try:
                engine.execute(matched_id, actor=actor, target=target)
                any_mechanic_fired = True
                mechanics_fired.append(matched_id)
            except Exception as e:
                tick_ctx.set_summary(
                    uc_id=uc_id,
                    outcome="error",
                    mechanics_fired=mechanics_fired,
                )
                pytest.fail(
                    f"{uc_id}: mechanic {matched_id!r} raised {type(e).__name__}: {e}"
                )

        # 4. Explicit outcome branching (W5) — record summary THEN branch.
        tick_ctx.set_summary(
            uc_id=uc_id,
            outcome=outcome,
            mechanics_fired=mechanics_fired,
        )

        if outcome == "yield":
            if not any_mechanic_fired:
                pytest.xfail(
                    f"{uc_id}: no matching mechanic — expected yield "
                    f"(resolve via authoring or Phase 5 classifier)"
                )
            # else fall through to assertions; UC is ready to flip to 'pass'
        elif outcome == "pass":
            if not any_mechanic_fired:
                pytest.fail(
                    f"{uc_id}: expected mechanic to match but none did"
                )
            # fall through to assertions
        # blocked was handled earlier

        # 5. Assert graph_assertions for pass + yield-that-fired cases
        for obs in fm.get("expected_observations") or []:
            for assertion in obs.get("graph_assertions") or []:
                _run_graph_assertion(harness_kg, assertion, uc_id)


def _run_graph_assertion(kg, assertion: dict, uc_id: str) -> None:
    kind = assertion.get("kind")
    if kind == "has_node":
        node = assertion["node"]
        assert kg.has_node(node), f"{uc_id}: expected node {node!r} missing"
    elif kind == "has_edge":
        src = assertion["src"]
        dst = assertion["dst"]
        rel = assertion.get("relation")
        assert kg.has_edge(src, dst, relation=rel) if rel else kg.has_edge(src, dst), \
            f"{uc_id}: expected edge {src}-{rel}->{dst} missing"
    elif kind == "not_has_edge":
        src = assertion["src"]
        dst = assertion["dst"]
        rel = assertion.get("relation")
        present = kg.has_edge(src, dst, relation=rel) if rel else kg.has_edge(src, dst)
        assert not present, f"{uc_id}: unexpected edge {src}-{rel}->{dst} present"
    elif kind == "has_property":
        node = assertion["node"]
        prop = assertion["property"]
        expected = assertion.get("value")
        props = kg.query(node)
        assert prop in props, f"{uc_id}: property {prop!r} missing on {node}"
        if expected is not None:
            assert props[prop] == expected, f"{uc_id}: {node}.{prop}={props[prop]!r} != {expected!r}"
    # Unknown assertion kinds are skipped (forward-compat)
```
