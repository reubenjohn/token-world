class WorldPropertyMatcher(Matcher):
    """Involuntary-mechanic matcher: triggers on world-level property changes.

    Closes GAP-ENG09. The `_world` sentinel node holds universe-scoped properties
    (season, weather, day_of_year). Mechanics that react to such changes declare
    this matcher in their ``watches()`` method.
    """
    def __init__(self, property_name: str) -> None:
        self.property_name = property_name

    def match(self, event) -> bool:
        # Existing event shape from Phase 1 EventStore
        if getattr(event, "event_type", None) != "set_property":
            return False
        if getattr(event, "node_id", None) != "_world":
            return False
        if getattr(event, "property_name", None) != self.property_name:
            return False
        return True


class DecayMatcher(Matcher):
    """Per-tick decay matcher: triggers on every node with a decay_period property.

    Consumed by Phase 5 passive sweep (D-17). Unlike event-based matchers, this
    is activated by the passive sweep directly iterating the graph for nodes
    that have the property; the match() method returns True only when invoked
    with a node reference (not an event) — passive sweep handles dispatch.
    """
    def match(self, event) -> bool:
        # DecayMatcher is not event-driven; always False in event-matching context.
        # Passive sweep iterates the graph directly.
        return False

    def matches_node(self, node_props: dict) -> bool:
        return "decay_period" in node_props


class TickMatcher(Matcher):
    """Fires once per tick unconditionally.

    Consumed by Phase 5 passive sweep (D-17). Used for world-state reactions
    that must run every tick (e.g., weather sampling via GAP-ENG09 hooks).
    """
    def match(self, event) -> bool:
        return False  # not event-driven; passive sweep dispatches directly
```

"""Deterministic matcher (D-09).

Iterates voluntary mechanics, scores each against a classified action, returns
the highest-scoring mechanic or NoMatchResult if no mechanic scores above zero.

Scoring formula:
    +3 if mechanic's VerbMatcher verb matches classified.verb
    +2 if target node type is acceptable to the mechanic (via its matchers)
    +1 if actor node type is acceptable

Ties broken alphabetically by mechanic id (stable, deterministic).
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from token_world.engine.models import (
    ClassifiedAction,
    MatchedResult,
    MatchResult,
    NoMatchResult,
)
from token_world.mechanic.matchers import VerbMatcher
from token_world.mechanic.protocol import Mechanic


def score_mechanic(mechanic: Mechanic, classified: ClassifiedAction, graph: Any) -> int:
    score = 0
    watches = mechanic.watches()
    # Verb match
    for m in watches:
        if isinstance(m, VerbMatcher) and m.verb == classified.verb:
            score += 3
            break
    # Target type (heuristic: any NodeMatcher with a node_type hint)
    if classified.target is not None and graph.has_node(classified.target):
        target_props = graph.query(classified.target) or {}
        target_type = target_props.get("type") or target_props.get("subtype")
        if target_type and _mechanic_accepts_type(mechanic, target_type, "target"):
            score += 2
    # Actor type
    if graph.has_node(classified.actor):
        actor_props = graph.query(classified.actor) or {}
        actor_type = actor_props.get("type") or actor_props.get("subtype")
        if actor_type and _mechanic_accepts_type(mechanic, actor_type, "actor"):
            score += 1
    return score


def _mechanic_accepts_type(mechanic: Mechanic, type_tag: str, role: str) -> bool:
    """Check whether the mechanic's tags / target hints include the given type.

    Heuristic: mechanic.tags OR a declared `{role}_types` attribute on the class.
    """
    tags = getattr(mechanic, "tags", []) or []
    if type_tag in tags:
        return True
    hints = getattr(mechanic, f"{role}_types", []) or []
    return type_tag in hints


@dataclass(slots=True)
class DeterministicMatcher:
    """D-09 matcher: scores voluntary mechanics, returns best or no_match."""

    def match(self, classified: ClassifiedAction, registry: Any, graph: Any) -> MatchResult:
        voluntary = registry.voluntary_mechanics()
        if not voluntary:
            return NoMatchResult(classified=classified, candidates=[])

        scored: list[tuple[str, int]] = [
            (m.id, score_mechanic(m, classified, graph)) for m in voluntary
        ]
        # Sort: descending score, ascending id (tie-break)
        scored.sort(key=lambda t: (-t[1], t[0]))
        top_id, top_score = scored[0]
        if top_score == 0:
            # No mechanic matched — candidates for debugging = all zero-score (limited)
            return NoMatchResult(classified=classified, candidates=[])

        # Candidates: next-best zero-scored mechanics would be empty; include
        # non-winning positively-scored mechanics so YieldSignal.candidate_mechanic_ids
        # has something useful if there's a near-tie but no-match case.
        candidates = [mid for mid, sc in scored[1:6] if sc > 0]

        reasoning = (
            f"clear winner (score {top_score})" if len(scored) < 2 or scored[1][1] < top_score
            else f"tie-break vs {scored[1][0]} (both score {top_score})"
        )

        return MatchedResult(
            mechanic_id=top_id,
            score=top_score,
            reasoning=reasoning,
        )
```
