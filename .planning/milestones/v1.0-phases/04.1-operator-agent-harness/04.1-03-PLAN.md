    ```python
    from __future__ import annotations

    import json
    import subprocess
    from pathlib import Path
    from typing import Any

    from claude_agent_sdk import create_sdk_mcp_server, tool


    def build_validation_server(universe: Path):
        """Returns an in-process SDK MCP server exposing validate_mechanic.

        Wraps Phase 4's `validate-mechanic --format json` CLI so the subagent
        sees structured results it can reason over (is_error flag + JSON findings)
        rather than parsing stdout text.
        """

        @tool(
            "validate_mechanic",
            (
                "Validate a mechanic module against the Phase 4 pipeline "
                "(syntax, AST rules, import, contract, tests, dry-execute). "
                "Returns structured report. Call after every edit to the file."
            ),
            {"module_path": str},
        )
        async def validate_mechanic(args: dict[str, Any]) -> dict[str, Any]:
            module_path = args["module_path"]
            result = subprocess.run(
                [
                    "uv", "run", "token-world", "validate-mechanic",
                    "--path", module_path,
                    "--format", "json",
                ],
                capture_output=True,
                text=True,
                cwd=str(universe),
                shell=False,  # T-04.1-11: never shell-interpolate
            )
            try:
                report = json.loads(result.stdout)
            except json.JSONDecodeError:
                return {
                    "content": [{
                        "type": "text",
                        "text": (
                            f"Validator crashed (returncode={result.returncode}).\n"
                            f"stdout: {result.stdout[:500]}\n"
                            f"stderr: {result.stderr[:500]}"
                        ),
                    }],
                    "is_error": True,
                }
            return {
                "content": [{
                    "type": "text",
                    "text": json.dumps(report, indent=2, sort_keys=True),
                }],
                "is_error": not bool(report.get("passed", False)),
            }

        return create_sdk_mcp_server(
            name="validation",
            version="1.0.0",
            tools=[validate_mechanic],
        )
    ```

    ```python
    from __future__ import annotations

    import json
    from pathlib import Path
    from typing import Any


    def load_universe_mcp_config(universe_path: Path) -> dict[str, Any]:
        """Parse a universe's .mcp.json into the Agent SDK's mcp_servers dict shape.

        Input file shape (Claude Code convention):
            {"mcpServers": {"token-world": {"command": "...", "args": [...]}}}

        Output shape (Agent SDK convention per RESEARCH Example 4):
            {"token-world": {"command": "...", "args": [...]}}
        """
        mcp_path = universe_path / ".mcp.json"
        if not mcp_path.exists():
            raise FileNotFoundError(
                f"No .mcp.json in {universe_path}. Is this a scaffolded universe?"
            )
        data = json.loads(mcp_path.read_text(encoding="utf-8"))
        servers = data.get("mcpServers", {})
        if not isinstance(servers, dict):
            raise ValueError(
                f"Malformed .mcp.json at {mcp_path}: mcpServers must be an object"
            )
        return dict(servers)
    ```
