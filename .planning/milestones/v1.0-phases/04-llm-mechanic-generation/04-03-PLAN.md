Per D-21/D-22/D-23/D-24/D-25 from 04-CONTEXT.md.

from __future__ import annotations

import json
import os
import re
import shutil
import tempfile
from contextlib import contextmanager
from datetime import date, datetime, timezone
from pathlib import Path
from typing import Any, Iterator

SCHEMA_VERSION = 1
_TMP_PATTERN = re.compile(r".*\.tmp\..+$")
_SAFE_ID_PATTERN = re.compile(r"[^A-Za-z0-9_.-]")

Step B — `_atomic_write_json(path: Path, obj: Any) -> None` implementing RESEARCH.md's pattern:
```python
def _atomic_write_json(path: Path, obj: Any) -> None:
    """Write JSON atomically via tempfile + os.replace.

    Crashes mid-write leave a `.tmp.*` file in the same directory (cleaned up
    at next DiagnosticsSink init).
    """
    path.parent.mkdir(parents=True, exist_ok=True)
    fd, tmp_path_str = tempfile.mkstemp(
        dir=path.parent, prefix=path.name + ".", suffix=".tmp"
    )
    tmp_path = Path(tmp_path_str)
    try:
        with os.fdopen(fd, "w", encoding="utf-8") as f:
            json.dump(obj, f, indent=2, sort_keys=True)
            f.flush()
            os.fsync(f.fileno())
        os.replace(tmp_path, path)
    except Exception:
        if tmp_path.exists():
            try:
                tmp_path.unlink()
            except OSError:
                pass
        raise
```

class DiagnosticsSink:
    """Filesystem substrate for per-tick + per-validation diagnostics."""

    def __init__(self, universe_dir: Path) -> None:
        self._universe_dir = universe_dir.resolve()
        self._root = (self._universe_dir / "diagnostics").resolve()
        self._root.mkdir(parents=True, exist_ok=True)
        self._sweep_tmp_files()

    def _sweep_tmp_files(self) -> None:
        """Remove leftover .tmp.* files from crashed atomic writes."""
        for entry in self._root.rglob("*.tmp.*"):
            if entry.is_file() and not entry.is_symlink():
                # Safety: confirm entry is under our root (resolve() hardening).
                try:
                    resolved = entry.resolve()
                    resolved.relative_to(self._root)
                except (ValueError, OSError):
                    continue
                try:
                    entry.unlink()
                except OSError:
                    pass

    @contextmanager
    def open_tick(self, tick_id: int) -> Iterator["TickDiagnostics"]:
        tick_dir = self._root / f"tick_{tick_id}"
        tick_dir.mkdir(exist_ok=True)
        ctx = TickDiagnostics(tick_dir=tick_dir, tick_id=tick_id)
        try:
            yield ctx
        finally:
            ctx.finalize()

    def open_validation(self, mechanic_id: str) -> Path:
        ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
        safe_id = _SAFE_ID_PATTERN.sub("_", mechanic_id) or "unknown"
        d = self._root / "validation" / f"{ts}_{safe_id}"
        d.mkdir(parents=True, exist_ok=True)
        # T-04-DIAG-PATH-TRAVERSAL: verify final path still under root
        resolved = d.resolve()
        resolved.relative_to(self._root)  # raises ValueError if outside
        return d

    def prune(self, *, before_tick: int | None = None,
              before_date: date | None = None,
              confirm: bool = False) -> list[Path]:
        """Remove tick_<id>/ and validation/*/ folders matching cutoff.

        Dry-run by default (confirm=False). Returns the list of paths that would
        be (or were) removed. Refuses to recurse into symlinks.

        T-04-PRUNE-DESTRUCTION: every candidate is verified to be under
        self._root via Path.resolve().relative_to before unlink.
        """
        if before_tick is None and before_date is None:
            raise ValueError("prune requires before_tick or before_date")

        candidates: list[Path] = []
        # Tick folders
        for entry in sorted(self._root.iterdir()) if self._root.is_dir() else []:
            if entry.is_symlink():
                continue
            name = entry.name
            if name.startswith("tick_") and entry.is_dir():
                try:
                    tid = int(name.removeprefix("tick_"))
                except ValueError:
                    continue
                if before_tick is not None and tid < before_tick:
                    candidates.append(entry)
                elif before_date is not None:
                    # Use folder mtime
                    mt = datetime.fromtimestamp(entry.stat().st_mtime, timezone.utc).date()
                    if mt < before_date:
                        candidates.append(entry)

        # Validation folders
        vroot = self._root / "validation"
        if vroot.is_dir():
            for entry in sorted(vroot.iterdir()):
                if entry.is_symlink() or not entry.is_dir():
                    continue
                # Timestamp prefix of form YYYYMMDDThhmmssZ_...
                name = entry.name
                try:
                    ts_part = name.split("_", 1)[0]
                    folder_dt = datetime.strptime(ts_part, "%Y%m%dT%H%M%SZ").replace(tzinfo=timezone.utc)
                except (ValueError, IndexError):
                    continue
                if before_date is not None and folder_dt.date() < before_date:
                    candidates.append(entry)

        # T-04-PRUNE-DESTRUCTION: verify every candidate is under root
        safe_candidates: list[Path] = []
        for c in candidates:
            try:
                c.resolve().relative_to(self._root)
            except (ValueError, OSError):
                continue
            safe_candidates.append(c)

        if confirm:
            for c in safe_candidates:
                shutil.rmtree(c, ignore_errors=False)

        return safe_candidates
```

class TickDiagnostics:
    """Per-tick diagnostics writer. Created by DiagnosticsSink.open_tick()."""

    def __init__(self, tick_dir: Path, tick_id: int) -> None:
        self._dir = tick_dir
        self.tick_id = tick_id
        self._summary: dict[str, Any] = {
            "schema_version": SCHEMA_VERSION,
            "tick_id": tick_id,
            "status": "pending",
        }
        self._finalized = False

    def write_action(self, text: str) -> None:
        (self._dir / "action.txt").write_text(text, encoding="utf-8")

    def write_classification(self, *, prompt: str, response: str, parsed: dict) -> None:
        d = self._dir / "classification"
        d.mkdir(exist_ok=True)
        (d / "prompt.txt").write_text(prompt, encoding="utf-8")
        (d / "response.txt").write_text(response, encoding="utf-8")
        _atomic_write_json(d / "parsed.json", parsed)

    def write_matching(self, matched: list[dict]) -> None:
        _atomic_write_json(self._dir / "matching.json", matched)

    def append_mutation(self, mutation_dict: dict) -> None:
        d = self._dir / "execution"
        d.mkdir(exist_ok=True)
        # JSON-lines append (per-line is unit of consistency)
        with (d / "mutations.jsonl").open("a", encoding="utf-8") as f:
            f.write(json.dumps(mutation_dict) + "\n")

    def write_execution_trace(self, trace_dict: dict) -> None:
        d = self._dir / "execution"
        d.mkdir(exist_ok=True)
        _atomic_write_json(d / "trace.json", trace_dict)

    def write_observation(self, *, prompt: str, response: str, parsed: dict) -> None:
        d = self._dir / "observation"
        d.mkdir(exist_ok=True)
        (d / "prompt.txt").write_text(prompt, encoding="utf-8")
        (d / "response.txt").write_text(response, encoding="utf-8")
        _atomic_write_json(d / "parsed.json", parsed)

    def set_summary(self, **fields: Any) -> None:
        self._summary.update(fields)

    def finalize(self) -> None:
        if self._finalized:
            return
        self._summary.setdefault("status", "ok")
        _atomic_write_json(self._dir / "summary.json", self._summary)
        self._finalized = True
```

    registry continues. Primary contract (indexing) must not be broken by a
    diagnostics write failure.

    # mechanic.diagnostics (both are exported from the mechanic package).
